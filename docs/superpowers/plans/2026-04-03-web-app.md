# TransitKit Web App — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Nuxt 3 web app mobile-first su Vercel con host-based routing multi-operatore, stop pages per QR code, lista linee, e script offline per generare i PNG dei QR.

**Architecture:** Single Vercel project con middleware Nuxt che legge l'header `Host` per selezionare l'operatore. Dati (config.json + schedules.json) serviti da GitHub Pages CDN. SSR + ISR (revalidate 3600s) via `routeRules` Nuxt — nessun `nuxt generate` upfront. Real-time GTFS-RT client-side con fallback silenzioso a dati statici.

**Tech Stack:** Nuxt 3, TypeScript, Tailwind CSS v4, Vitest, protobufjs (GTFS-RT), qrcode (QR PNG)

  ```ts
  import type { ScheduleData, Departure, DayGroup } from '~/types'

  const WEEKDAY_ABBR = ['sun', 'mon', 'tue', 'wed', 'thu', 'fri', 'sat'] as const
  const WEEKDAY_LABELS: Record<string, string> = {
    mon: 'Lun', tue: 'Mar', wed: 'Mer', thu: 'Gio',
    fri: 'Ven', sat: 'Sab', sun: 'Dom',
  }

  export function parseDayGroup(key: string): DayGroup {
    const days = key.split(',').map(d => d.trim())
    return {
      id: key,
      days,
      displayLabel: buildDisplayLabel(days),
    }
  }

  function buildDisplayLabel(days: string[]): string {
    if (days.length === 7) return 'Ogni giorno'
    if (days.length === 5 && days.every(d => !['sat', 'sun'].includes(d))) return 'Lun–Ven'
    if (days.length === 2 && days.includes('sat') && days.includes('sun')) return 'Sab–Dom'
    if (days.length === 1) return WEEKDAY_LABELS[days[0]] ?? days[0]
    return days.map(d => WEEKDAY_LABELS[d] ?? d).join(', ')
  }

  export function getTodayDayGroupKey(
    departures: Record<string, (string | number)[][]>,
  ): string | null {
    const todayAbbr = WEEKDAY_ABBR[new Date().getDay()]
    for (const key of Object.keys(departures)) {
      if (key.split(',').includes(todayAbbr)) return key
    }
    return null
  }

  export function decodeDepartures(
    compactDeps: (string | number)[][],
    data: ScheduleData,
    headsignMap?: Record<string, string>,
  ): Departure[] {
    const result: Departure[] = []

    for (const compact of compactDeps) {
      if (compact.length < 3) continue

      const time = String(compact[0])
      const lineIdx = Number(compact[1])
      const headsignIdx = Number(compact[2])

      if (
        lineIdx >= data.lineNames.length ||
        lineIdx >= data.routeIds.length ||
        headsignIdx >= data.headsigns.length
      ) continue

      const lineName = data.lineNames[lineIdx]!
      const routeId = data.routeIds[lineIdx]!
      const rawHeadsign = data.headsigns[headsignIdx]!
      const headsign = headsignMap?.[rawHeadsign] ?? rawHeadsign

      const dock = compact.length > 3 && typeof compact[3] === 'string' ? compact[3] : ''
      const tripIdIdx = compact.length > 5 ? Number(compact[5]) : undefined
      const tripId = tripIdIdx !== undefined && tripIdIdx < data.tripIds.length
        ? data.tripIds[tripIdIdx]
        : undefined

      const route = data.routes.find(r => r.id === routeId)

      const [hStr, mStr] = time.split(':')
      const h = parseInt(hStr ?? '0', 10)
      const m = parseInt(mStr ?? '0', 10)

      result.push({
        id: `${time}_${lineName}_${headsign}_${dock}`,
        time,
        lineName,
        routeId,
        headsign,
        color: route?.color ?? '#888888',
        textColor: route?.textColor ?? '#FFFFFF',
        transitType: route?.transitType ?? 'bus',
        dock,
        tripId,
        minutesFromMidnight: h * 60 + m,
      })
    }

    return result
  }
  ```

  ```ts
  import type { Departure } from '~/types'

  /**
   * Merge puro — applicato sopra i departures statici dopo ogni poll GTFS-RT.
   * Esportato per essere testabile in isolamento.
   */
  export function mergeRealtimeDelays(
    departures: Departure[],
    delays: Record<string, number>, // tripId → delay in secondi
  ): Departure[] {
    return departures.map(dep => {
      if (!dep.tripId || !(dep.tripId in delays)) return dep
      return {
        ...dep,
        realtimeDelay: delays[dep.tripId],
        isRealtime: true,
      }
    })
  }

  /**
   * Composable client-side per GTFS-RT.
   * Fa polling ogni 30s, fallback silenzioso se CORS o fetch fallisce.
   *
   * @param departures - Ref di departures statici (input)
   * @param gtfsRtUrl  - URL trip_updates protobuf (può essere undefined)
   */
  export function useRealtime(
    departures: Ref<Departure[]>,
    gtfsRtUrl: string | undefined,
  ) {
    const isLive = ref(false)
    const merged = ref<Departure[]>(departures.value)

    // Sync iniziale
    watch(departures, (deps) => {
      merged.value = isLive.value
        ? merged.value // mantieni overlay realtime
        : deps
    })

    if (!gtfsRtUrl || import.meta.server) {
      return { departures: merged, isLive }
    }

    async function poll() {
      try {
        // fetch protobuf binario
        const res = await fetch(gtfsRtUrl, { mode: 'cors' })
        if (!res.ok) throw new Error(`HTTP ${res.status}`)
        const buffer = await res.arrayBuffer()

        // parse con protobufjs
        const protobuf = await import('protobufjs')
        const root = await protobuf.load('/proto/gtfs-realtime.proto')
        const FeedMessage = root.lookupType('transit_realtime.FeedMessage')
        const feed = FeedMessage.decode(new Uint8Array(buffer)) as unknown as GtfsRtFeed

        // Costruisci mappa tripId → delay
        const delays: Record<string, number> = {}
        for (const entity of feed.entity ?? []) {
          const tu = entity.tripUpdate
          if (!tu?.trip?.tripId) continue
          const delay = tu.stopTimeUpdate?.[0]?.departure?.delay
            ?? tu.stopTimeUpdate?.[0]?.arrival?.delay
            ?? 0
          delays[tu.trip.tripId] = delay
        }

        merged.value = mergeRealtimeDelays(departures.value, delays)
        isLive.value = true
      } catch {
        // CORS, 404, parse error — degrada silenziosamente
        isLive.value = false
        merged.value = departures.value
      }
    }

    // Poll iniziale + intervallo
    let timer: ReturnType<typeof setInterval>
    onMounted(async () => {
      await poll()
      timer = setInterval(poll, 30_000)
    })
    onUnmounted(() => clearInterval(timer))

    return { departures: merged, isLive }
  }

  // Minimal GTFS-RT types per TypeScript
  interface GtfsRtFeed {
    entity?: Array<{
      tripUpdate?: {
        trip?: { tripId?: string }
        stopTimeUpdate?: Array<{
          departure?: { delay?: number }
          arrival?: { delay?: number }
        }>
      }
    }>
  }
  ```

**Files:**
- Create: `scripts/generate-qrs.ts`

- [ ] **Step 1: Verifica dipendenze root**

- [ ] **Step 2: Crea lo script**

  Crea `scripts/generate-qrs.ts`:
  ```ts
  /**
   * Genera QR code PNG per ogni fermata di un operatore.
   *
   * Uso:
   *   npx tsx scripts/generate-qrs.ts --operator appalcart
   *   npx tsx scripts/generate-qrs.ts --operator appalcart --host fermate.appalcart.com
   *
   * Output: qr/{operatorId}/stop-{stopId}-{stopName}.png
   *         qr/{operatorId}/qr-codes.zip
   */

  import QRCode from 'qrcode'
  import { promises as fs } from 'fs'
  import path from 'path'
  import { createWriteStream } from 'fs'

  // Parse CLI args
  const args = process.argv.slice(2)
  const operatorArg = args[indexOf('--operator', args) + 1]
  const hostArg = args[indexOf('--host', args) + 1]

  function indexOf(flag: string, arr: string[]): number {
    const i = arr.indexOf(flag)
    return i === -1 ? arr.length : i
  }

  if (!operatorArg) {
    console.error('Usage: npx tsx scripts/generate-qrs.ts --operator <operatorId>')
    process.exit(1)
  }

  const OPERATOR_ID = operatorArg
  const CDN_BASE = process.env.CDN_BASE ?? 'https://YOUR_GITHUB_USER.github.io/transitkit-data'
  const WEB_HOST = hostArg ?? `${OPERATOR_ID}.transitkit.app`
  const OUTPUT_DIR = path.join(process.cwd(), 'qr', OPERATOR_ID)

  function slugify(name: string): string {
    return name
      .toLowerCase()
      .replace(/[^a-z0-9]+/g, '-')
      .replace(/^-|-$/g, '')
  }

  async function main() {
    // 1. Scarica schedules.json dal CDN
    console.log(`Downloading schedules from ${CDN_BASE}/${OPERATOR_ID}/schedules.json ...`)
    const res = await fetch(`${CDN_BASE}/${OPERATOR_ID}/schedules.json`)
    if (!res.ok) throw new Error(`CDN fetch failed: ${res.status}`)
    const data = await res.json() as { stops: Array<{ id: string; name: string }> }

    // 2. Crea output directory
    await fs.mkdir(OUTPUT_DIR, { recursive: true })

    // 3. Genera un QR per ogni fermata
    let generated = 0
    for (const stop of data.stops) {
      const url = `https://${WEB_HOST}/stop/${stop.id}`
      const filename = `stop-${stop.id}-${slugify(stop.name)}.png`
      const filepath = path.join(OUTPUT_DIR, filename)

      await QRCode.toFile(filepath, url, {
        type: 'png',
        width: 300,
        margin: 2,
        errorCorrectionLevel: 'H',  // Alta resistenza a danni fisici
        color: {
          dark: '#000000',
          light: '#FFFFFF',
        },
      })

      generated++
      if (generated % 50 === 0) {
        console.log(`  ${generated}/${data.stops.length} fermate...`)
      }
    }

    console.log(`\n✓ ${generated} QR code generati in ${OUTPUT_DIR}/`)
    console.log(`  URL formato: https://${WEB_HOST}/stop/{stopId}`)
    console.log(`\nConsegna la cartella ${OUTPUT_DIR}/ all'operatore (zip o Google Drive).`)
  }

  main().catch(err => {
    console.error('Errore:', err)
    process.exit(1)
  })
  ```
